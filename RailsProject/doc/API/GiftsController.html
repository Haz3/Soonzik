def save
  begin
    if (@security)
      gift = Gift.new(@gift)
      classObj = gift.typeObj.constantize
      # check if the object exists
      if (classObj.find_by_id(gift.obj_id) != nil && gift.save)
        @returnValue = { content: gift.as_json(:include => {
                                                                :user_from => {},
                                                                :user_to => {}
                                                              }) }
        codeAnswer 201
      else
        codeAnswer 503
      end
    else
      codeAnswer 500
    end
  rescue
    codeAnswer 504
  end
  sendJson
end
